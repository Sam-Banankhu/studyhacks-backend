User Management Endpoints

POST /register

Register a new user.

                    
                        Request Body:
                        {
                            "email": "user@example.com",
                            "name": "John Doe",
                            "role": "user",
                            "profile_complete": true,
                            "profile_photo": "https://example.com/photo.jpg",
                            "password": "password"
                        }
                        Response:
                        {
                            "msg": "User registered successfully"
                        }

POST /login

Authenticate and log in a user, providing an access token.

                    
                        Request Body:
                        {
                            "email": "user@example.com",
                            "password": "password"
                        }
                        Response:
                        {
                            "msg": "Login successful",
                            "access_token": "your_access_token",
                            "user": {
                                "user_id": "user_id",
                                "profile_complete": true,
                                "profile_photo": "https://example.com/photo.jpg",
                                "email": "user@example.com"
                            }
                        }

GET /users

Get a list of all users.

                    
                        Response:
                        [
                            {
                                "user_id": "user_id",
                                "name": "John Doe",
                                "email": "user@example.com",
                                "role": "user",
                                "profile_complete": true,
                                "profile_photo": "https://example.com/photo.jpg"
                            },
                            // More user objects...
                        ]

POST /logout

Log out the current user.

                    
                        Response:
                        {
                            "msg": "Logged out successfully"
                        }

GET /users/<user_id>

Get user details by user ID.

                    
                        Response:
                        {
                            "user_id": "user_id",
                            "name": "John Doe",
                            "email": "user@example.com",
                            "role": "user",
                            "profile_complete": true,
                            "profile_photo": "https://example.com/photo.jpg"
                        }

POST /change_password

Change the user's password.

                    
                        Request Body:
                        {
                            "current_password": "current_password",
                            "new_password": "new_password"
                        }
                        Response:
                        {
                            "msg": "Password updated successfully"
                        }

PDF Processing Endpoints

POST /pdfs

Upload a PDF file for processing.

                    
                        Request Form data:
                        {
                            "file": "your file here",
                            "new_password": "new_password"
                        }
                        Request Header:
                        {
                            "Authorization": "Bearer your_access_token"
                        }
                        Response:
                        {
                            "msg": "Document processed successfully"
                        }

GET /pdfs

Get a list of all processed PDF files.

                    
                        Request Header:
                        {
                            "Authorization": "Bearer your_access_token"
                        }
                        Response:
                        [
                            {
                                "user_id": "user_id",
                                "pdf_name": "document1.pdf",
                                "chat_ids": [],
                                "timestamp": "2023-09

-14T12:34:56.789Z",
                                "extracted_text": "This is the extracted text."
                            },
                            // More PDF objects...
                        ]

GET /pdfs/download/<pdf_id>

Download a processed PDF file.

                    
                        
                        Request Header:
                        {
                            "Authorization": "Bearer your_access_token"
                        }

Chat Processing Endpoints

POST /chats/<pdf_id>

Create a chat based on a PDF document.

                    
                        Request Header:
                        {
                            "Authorization": "Bearer your_access_token"
                        }
                        Request Body:
                        {
                            "question": "ask a question here?"
                        }
                        Response:
                        {
                            "msg": "Chat created successfully",
                            "question": "question",
                            "answer": "The answer is here",
                            "pdf_id": "pdf_id"
                        }

GET /chats

Get a list of all chats created by the current user.

                    
                        Request Header:
                        {
                            "Authorization": "Bearer your_access_token"
                        }
                        Response:
                        [
                            {
                                "_id": "chat_id",
                                "question": "What is the answer?",
                                "answer": "The answer is...",
                                "pdf_id": "pdf_id",
                                "timestamp": "2023-09-14T12:34:56.789Z"
                            },
                            
                        ]

GET /chats/<chat_id>

Get a specific chat by ID.

                    
                        Request Header:
                        {
                            "Authorization": "Bearer your_access_token"
                        }
                        Response:
                        {
                            "chat": {
                                "_id": "chat_id",
                                "question": "What is the answer?",
                                "answer": "The answer is...",
                                "pdf_id": "pdf_id",
                                "timestamp": "2023-09-14T12:34:56.789Z"
                            }
                        }

PUT /chats/<chat_id>

Update a specific chat by ID.

                    
                        Request Header:
                        {
                            "Authorization": "Bearer your_access_token"
                        }
                        Request Body:
                        {
                            // Updated chat data
                        }
                        Response:
                        {
                            "msg": "Chat updated successfully"
                        }

DELETE /chats/<chat_id>

Delete a specific chat by ID.

                    
                        Request Header:
                        {
                            "Authorization": "Bearer your_access_token"
                        }
                        Response:
                        {
                            "msg": "Chat deleted successfully"
                        }
